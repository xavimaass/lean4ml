\chapter{General Definitions and Basic Properties}\label{sec:GeneralDefinitions}

Throughout this section, $E$ is a real Hilbert space, $f : E \to \mathbb{R}$,
$s \subseteq E$, and $L \in \mathbb{R}_{\ge 0}$ (Lean: \texttt{NNReal}).

Many of the properties presented here are well-known and we don't give an explicit proof.
Still, these were added to our Lean project to simplify the proofs of more complex results.

\section{L-smoothness and others}
\begin{definition}[$L$-smoothness on a set]\label{def:l_smooth_on}
  \lean{Lean4ML.Optimization.LSmoothOn}
  \leanok{}
  \uses{}
  We say that $f$ is $L$-smooth on $s$ if
  \[
    \|\nabla f(x) - \nabla f(y)\| \le L\|x-y\|,\quad \forall x,y\in s.
  \]
  %In Lean:
  %\texttt{LSmoothOn f L s~:= LipschitzOnWith L (fun x => gradient f x) s}.
\end{definition}

\begin{definition}[Global $L$-smoothness]\label{def:l_smooth_fn}
  \lean{Lean4ML.Optimization.LSmoothfn}
  \leanok{}
  \uses{}
  A function is globally $L$-smooth if it is $L$-smooth on all of $E$.%$\nabla f$ is $L$-Lipschitz on all of $E$.
  %In Lean:
  %\texttt{LSmoothfn f L~:= LipschitzWith L (fun x => gradient f x)}.
\end{definition}

\begin{lemma}[Global smoothness is smoothness on \texttt{Set.univ}]\label{lem:lSmooth_iff_lSmoothOn_univ}
  \lean{Lean4ML.Optimization.lSmooth_iff_lSmoothOn_univ}
  \leanok{}
  \uses{def:l_smooth_on,def:l_smooth_fn}
  In Lean:
  \[
    \texttt{LSmoothfn f L} \;\Longleftrightarrow\; \texttt{LSmoothOn f L Set.univ}.
  \]
  %This is a \texttt{simp} lemma using \texttt{lipschitzOnWith\_univ}.
\end{lemma}

\begin{lemma}[Monotonicity under set inclusion]\label{lem:LSmoothOn_mono}
  \lean{Lean4ML.Optimization.LSmoothOn_mono}
  \leanok{}
  \uses{def:l_smooth_on}
  If $T \subseteq S$ and $f$ is $L$-smooth on $S$, then $f$ is $L$-smooth on $T$.
\end{lemma}

\begin{definition}[Hessian]\label{def:hessian}
  \lean{Lean4ML.Optimization.hessian}
  \leanok{}
  \uses{}
  The Hessian is the Fr\'echet derivative of the gradient map:
  \[
    \nabla^2 f(x) := D(\nabla f)(x).
  \]
  %In Lean:\texttt{hessian f x := fderiv ℝ (fun y => gradient f y) x}.
\end{definition}


\begin{definition}[Globally $\mu$-PL function]\label{def:GloballyPL}
  \lean{Lean4ML.Optimization.GloballyPL}
  \leanok{}
  \uses{}
  We say that $f$ is globally $\mu$-PL if for all $x$ we have
\[
  \|\nabla f(x)\|^2 \ge 2\mu (f(x) - \inf f).
\]

\end{definition}

\section{Line Map and its properties}
\begin{definition}[Line map]\label{def:lineMap}
  \lean{Lean4ML.Optimization.lineMap}
  \leanok{}
  \uses{}
  The canonical line map through $x$ in direction $p$ is
  \(
    t \mapsto x + t\,p.

\end{definition}

\begin{lemma}[Continuity of $t \mapsto (Df(x+t p))p$]\label{lem:continuous_fderiv_line_apply}
  \lean{Lean4ML.Optimization.continuous_fderiv_line_apply}
  \leanok{}
  \uses{}

\end{lemma}

\begin{lemma}[Continuity of $t \mapsto \langle \nabla f(x+t p),p\rangle$]\label{lem:continuous_inner_gradient_line}
  \lean{Lean4ML.Optimization.continuous_inner_gradient_line}
  \leanok{}
  \uses{lem:hasGradientAt_of_contDiff_one,lem:continuous_fderiv_line_apply}
  If $f$ is $C^1$, then $t \mapsto \langle \nabla f(x+t p),p\rangle$ is continuous.
\end{lemma}
\begin{proof}
  In Lean, simply rewrite $(Df(x+t p))p$ as $\langle \nabla f(x+t p),p\rangle$ using
  \texttt{HasGradientAt.fderiv\_apply}.
\end{proof}

\begin{lemma}[Integrability of the line integrand]\label{lem:intervalIntegrable_inner_gradient_line}
  \lean{Lean4ML.Optimization.intervalIntegrable_inner_gradient_line}
  \leanok{}
  \uses{lem:continuous_inner_gradient_line}
  The map \(t ↦ ⟪∇f(x + t p), p⟫\) is integrable on any interval (by continuity).
\end{lemma}

\begin{lemma}[Chain rule along a line]\label{lem:hasDerivAt_comp_line}
  \lean{Lean4ML.Optimization.hasDerivAt_comp_line}
  \leanok{}
  \uses{}
  If $f$ has Fr\'echet derivative at $x+t p$, then $\tau\mapsto f(x+\tau p)$ has derivative
  $(Df(x+t p))p$ at $\tau=t$.
\end{lemma}

\begin{lemma}[Convex segment stays in a convex set]\label{lem:Convex.add_smul_sub_mem_unit}
  \lean{Lean4ML.Optimization.Convex.add_smul_sub_mem_unit}
  \leanok{}
  \uses{}
  If $s$ is convex and $x,y\in s$, then $x+t(y-x)\in s$ for $t\in[0,1]$.
\end{lemma}

\begin{lemma}[Scaling of the Hessian quadratic form]\label{lem:hessian_inner_smul}

  \uses{def:hessian}
  \[
    \langle \nabla^2 f(x)(\alpha p),\alpha p\rangle = \alpha^2\langle \nabla^2 f(x)p,p\rangle.
  \]
\end{lemma}

\begin{lemma}[Hessian continuity along a line]\label{lem:continuous_hessian_line}
  \lean{Lean4ML.Optimization.continuous_hessian_line}
  \leanok{}
  \uses{def:hessian,lem:continuous_fderiv_line_apply}
  If $f\in C^2$, then $t\mapsto \nabla^2 f(x+t p)\,p$ is continuous.
\end{lemma}

\begin{theorem}[First-order Taylor formula along a line (vector-valued)]\label{thm:taylor_first_order_vector}
  \lean{Lean4ML.Optimization.taylor_first_order_vector}
  \leanok{}
  \uses{lem:continuous_fderiv_line_apply,lem:hasDerivAt_comp_line}
  If $f:E\to F$ is $C^1$, then
  \[
    f(x+p)-f(x) = \int_0^1 (Df(x+t p))p\,dt.
  \]
  %Lean uses \texttt{intervalIntegral.integral\_eq\_sub\_of\_hasDerivAt}.
\end{theorem}
\begin{proof}
  Let
  \(
    g(t) := f(x+t\,p)\quad (t\in\mathbb{R}).
  \)
  We apply the fundamental theorem of calculus to $g$ on $[0,1]$.

  For each $t$, since $f\in C^1$, $f$ is Fr\'echet differentiable at $x+t\,p$, with derivative
  $Df(x+t\,p)$. By the chain rule along a line (Lemma~\ref{lem:hasDerivAt_comp_line}),
  \(
    g'(t)=(Df(x+t\,p))p.
  \)
  Hence $g$ has the required derivative at every point of $\mathrm{uIcc}(0,1)$.

  Next, the integrand $t\mapsto (Df(x+t\,p))p$ is interval-integrable on $[0,1]$ by
  continuity (Lemma~\ref{lem:continuous_fderiv_line_apply}).

  Therefore, by
  \texttt{intervalIntegral.integral\_eq\_sub\_of\_hasDerivAt},
  \[
    \int_0^1 (Df(x+t\,p))p\,dt = g(1)-g(0).
  \]
  Expanding $g(1)=f(x+p)$ and $g(0)=f(x)$ gives
  \[
    f(x+p)-f(x)=\int_0^1 (Df(x+t\,p))p\,dt.
  \]
\end{proof}

\begin{theorem}[First-order Taylor formula along a line]\label{thm:taylor_first_order}
  \lean{Lean4ML.Optimization.taylor_first_order}
  \leanok{}
  \uses{thm:taylor_first_order_vector,lem:hasGradientAt_of_contDiff_one}
  If $f\in C^1$, then
  \[
    f(x+p)-f(x) = \int_0^1 \langle \nabla f(x+t p),p\rangle\,dt.
  \]
\end{theorem}
\begin{proof}
  Direct from Theorem~\ref{thm:taylor_first_order_vector}.
\end{proof}

\begin{theorem}[Second-order Taylor formula for the gradient]\label{thm:taylor_second_order_gradient}
  \lean{Lean4ML.Optimization.taylor_second_order_gradient}
  \leanok{}
  \uses{def:hessian,thm:taylor_first_order_vector}
  If $f\in C^2$, then
  \[
    \nabla f(x+p)-\nabla f(x)=\int_0^1 \nabla^2 f(x+t p)\,p\,dt.
  \]
\end{theorem}
\begin{proof}
  Direct from Theorem~\ref{thm:taylor_first_order_vector} applied to $\nabla f \in C^1$.
\end{proof}

\begin{lemma}[Pointwise Lipschitz-gradient bound]\label{lem:lipschitz_gradient_pointwise}
  \lean{Lean4ML.Optimization.lipschitz_gradient_pointwise}
  \leanok{}
  \uses{def:l_smooth_on,lem:norm_gradient_sub_le,lem:norm_add_smul_sub_eq}
  For $f\in C^1$ that is $L$-smooth, for $\gamma\ge 0$ and the segment point in $s$,
  \[
    \left\langle \nabla f(x+\gamma(y-x))-\nabla f(x),y-x\right\rangle
    \le L\gamma\|y-x\|^2.
  \]
\end{lemma}
\begin{proof}
  First, derive a norm bound from $L$-smoothness. Since $x\in s$ and
  $x+\gamma(y-x)\in s$, Lemma~\ref{lem:norm_gradient_sub_le} gives
  \[
    \|\nabla f(x+\gamma(y-x)) - \nabla f(x)\|
    \le L\,\|(x+\gamma(y-x))-x\|.
  \]
  Using Lemma~\ref{lem:norm_add_smul_sub_eq} (with $\gamma\ge 0$),
  \(
    \|(x+\gamma(y-x))-x\| = \gamma\|y-x\|,
  \)
  hence
  \(
    \|\nabla f(x+\gamma(y-x)) - \nabla f(x)\|
    \le L\gamma\|y-x\|.
  \)

  Second, apply Cauchy-Schwarz:
  \[
    \langle \nabla f(x+\gamma(y-x)) - \nabla f(x),y-x\rangle
    \le \|\nabla f(x+\gamma(y-x)) - \nabla f(x)\|\,\|y-x\|.
  \]
  Plug the previous bound into the right-hand side to get
  \[
    \langle \nabla f(x+\gamma(y-x)) - \nabla f(x),y-x\rangle
    \le (L\gamma\|y-x\|)\|y-x\| = L\gamma\|y-x\|^2,
  \]
  which is exactly the claim.
\end{proof}

\begin{lemma}[Non-parametric corollary ($\gamma=1$)]\label{lem:inner_grad_diff_le_L}
  \lean{Lean4ML.Optimization.inner_grad_diff_le_L}
  \leanok{}
  \uses{lem:lipschitz_gradient_pointwise}
  In particular, if $x,y\in s$ then
  $\langle \nabla f(y)-\nabla f(x),y-x\rangle\le L\|y-x\|^2$.
\end{lemma}

\begin{theorem}[Quadratic upper bound from smoothness on a convex set]\label{thm:l_smooth_quadratic_upper_bound}
  \lean{Lean4ML.Optimization.l_smooth_quadratic_upper_bound}
  \leanok{}
  \uses{thm:taylor_first_order,lem:lipschitz_gradient_pointwise,lem:Convex.add_smul_sub_mem_unit,lem:integral_inner_sub_const,lem:intervalIntegral_mono_of_continuous}
  If $f\in C^1$, $f$ is $L$-smooth on convex $s$, and $x,y\in s$, then
  \[
    f(y)\le f(x)+\langle \nabla f(x),y-x\rangle+\frac{L}{2}\|y-x\|^2.
  \]
\end{theorem}

\begin{proof}
  Let $p:=y-x$.
  \begin{enumerate}
    \item Apply Theorem~\ref{thm:taylor_first_order} to obtain
    $f(y)-f(x)=\int_0^1\langle \nabla f(x+t p),p\rangle\,dt$.
    \item Rewrite
    $f(y)-f(x)-\langle \nabla f(x),p\rangle$
    as an integral of
    $\langle \nabla f(x+t p)-\nabla f(x),p\rangle$
    using Lemma~\ref{lem:integral_inner_sub_const}.
    \item Use convexity via Lemma~\ref{lem:Convex.add_smul_sub_mem_unit} to ensure $x+t p\in s$ for $t\in[0,1]$.
    \item Bound the integrand pointwise using Lemma~\ref{lem:lipschitz_gradient_pointwise} with $\gamma=t$,
    then integrate the bound with Lemma~\ref{lem:intervalIntegral_mono_of_continuous}.
    \item Evaluate $\int_0^1 Lt\,dt=L/2$ and conclude.%using \texttt{integral\_id} and close by \texttt{linarith}.
  \end{enumerate}
\end{proof}

\begin{theorem}[Descent Lemma for Gradient Descent]\label{thm:gradient_descent_one_step}
  \lean{Lean4ML.Optimization.gradient_descent_one_step}
  \leanok{}
  \uses{thm:l_smooth_quadratic_upper_bound}
  If $f \in C^1$, $f$ is $L$-smooth on a convex set $s$, $x \in s$, $\alpha \geq 0$, and
  $x - \alpha \nabla f(x) \in s$, then
  \[
    f(x - \alpha \nabla f(x)) \le f(x) - \alpha\left(1 - \frac{L\alpha}{2}\right) \|\nabla f(x)\|^2.
  \]
\end{theorem}
\begin{proof}
  Let $y := x - \alpha \nabla f(x)$ and $p := y - x = -\alpha \nabla f(x)$.

  \textbf{Step 1:} Compute the inner product.
  \[
    \langle \nabla f(x), p \rangle = \langle \nabla f(x), -\alpha \nabla f(x) \rangle = -\alpha \|\nabla f(x)\|^2.
  \]

  \textbf{Step 2:} Compute the norm squared of $p$ using $\alpha \ge 0$ to simplify $|\alpha| = \alpha$.
  \[
    \|p\|^2 = \|-\alpha \nabla f(x)\|^2 = (\|\alpha \nabla f(x)\|)^2 = \alpha^2 \|\nabla f(x)\|^2.
  \]

  \textbf{Step 3:} Apply Theorem~\ref{thm:l_smooth_quadratic_upper_bound} with the points $x$ and $y = x + p$:
  \[
    f(y) \le f(x) + \langle \nabla f(x), p \rangle + \frac{L}{2} \|p\|^2.
  \]

  \textbf{Step 4:} Substitute the expressions from Steps 1 and 2:
  \begin{align*}
    f(y) &\le f(x) + (-\alpha \|\nabla f(x)\|^2) + \frac{L}{2}(\alpha^2 \|\nabla f(x)\|^2) \\
    &= f(x) - \alpha \|\nabla f(x)\|^2 + \frac{L\alpha^2}{2} \|\nabla f(x)\|^2 \\
    &= f(x) - \alpha\left(1 - \frac{L\alpha}{2}\right) \|\nabla f(x)\|^2.
  \end{align*}
  This completes the proof.
\end{proof}

\chapter{Hessian characterization of L-Smooth functions}\label{sec:HessianVsSmoothness}

\begin{theorem}[Hessian operator-norm bound $\Rightarrow$ global smoothness]\label{thm:hessianBoundImpliesLSmoothOnUniv}
  \lean{Lean4ML.Optimization.hessianBoundImpliesLSmoothOnUniv}
  \leanok{}
  \uses{def:l_smooth_fn,def:hessian}
  If $\nabla f$ is differentiable everywhere with derivative $\nabla^2 f(x)$, and $\|\nabla^2 f(x)\|\le L$
  for all $x$, then $f$ is globally $L$-smooth.
\end{theorem}

\begin{proof}
  We need to show $\nabla f$ is $L$-Lipschitz globally.

  Since $\nabla f$ is Fr\'echet differentiable at every point $x$ with derivative $\nabla^2 f(x)$
  (by hypothesis), and we have the operator-norm bound $\|\nabla^2 f(x)\| \le L$ for all $x$,
  Mathlib's \texttt{lipschitzWith\_of\_nnnorm\_fderiv\_le} directly gives that
  $\nabla f$ is $L$-Lipschitz on all of $E$. By definition, this means $f$ is globally $L$-smooth.
\end{proof}

\begin{theorem}[Global smoothness $\Rightarrow$ Hessian quadratic-form upper bound]\label{thm:lSmoothOnUnivImpliesHessianQuadraticFormBound}
  \lean{Lean4ML.Optimization.lSmoothOnUnivImpliesHessianQuadraticFormBound}
  \leanok{}
  \uses{def:l_smooth_fn,def:hessian,lem:lSmooth_iff_lSmoothOn_univ,lem:opNorm_fderiv_le_of_lipschitz,lem:inner_le_opNorm_mul_norm_sq}
  If $f$ is globally $L$-smooth, then
  $\langle \nabla^2 f(x)p,p\rangle \le L\|p\|^2$ for all $x,p$.
\end{theorem}

\begin{proof}
  Fix $x$ and $p$. By global $L$-smoothness, $\nabla f$ is $L$-Lipschitz on all of $E$.

  Apply Lemma~\ref{lem:opNorm_fderiv_le_of_lipschitz} to the gradient map: since $\nabla f$ is
  $L$-Lipschitz on a neighborhood of $x$ (in fact, globally) and is Fr\'echet differentiable at $x$
  with derivative $\nabla^2 f(x)$, we obtain
  \[
    \|\nabla^2 f(x)\| \le L.
  \]

  By Lemma~\ref{lem:inner_le_opNorm_mul_norm_sq} (operator norm controls the quadratic form),
  \[
    \langle \nabla^2 f(x) p, p \rangle \le \|\nabla^2 f(x)\| \cdot \|p\|^2 \le L \|p\|^2,
  \]
  as required.
\end{proof}

\chapter{NecessaryConditions}\label{sec:NecessaryConditions}

\begin{theorem}[Local minimum $\Rightarrow$ gradient vanishes]\label{thm:local_min_gradient_zero}
  \lean{Lean4ML.Optimization.local_min_gradient_zero}
  \leanok{}
  \uses{lem:hasGradientAt_of_contDiff_one}
  If $f\in C^1$ and $x$ is a local minimizer (Lean: \texttt{IsLocalMin f x}), then $\nabla f(x)=0$.
\end{theorem}

\begin{proof}
  By Lemma~\ref{lem:hasGradientAt_of_contDiff_one}, the gradient $\nabla f(x)$ exists at the local minimizer $x$.

  Since $x$ minimizes $f$ locally, the Fr\'echet derivative of $f$ at $x$ must vanish when applied to any direction.
  The Fr\'echet derivative of $f$ at $x$ is the inner product with $\nabla f(x)$, so we have
  \[
    \langle \nabla f(x), v \rangle = 0 \quad \text{for all } v \in E.
  \]
  Setting $v = \nabla f(x)$ gives $\|\nabla f(x)\|^2 = 0$, hence $\nabla f(x) = 0$.
\end{proof}

\begin{theorem}[Local minimum $\Rightarrow$ Hessian PSD]\label{thm:local_min_hessian_psd}
  \lean{Lean4ML.Optimization.local_min_hessian_psd}
  \leanok{}
  \uses{thm:local_min_gradient_zero,lem:taylor_second_order_exact,lem:continuous_hessian_line}
  If $f\in C^2$ and $x$ is a local minimizer, then $\langle \nabla^2 f(x)p,p\rangle\ge 0$ for all $p$.
\end{theorem}

\begin{proof}
  We proceed by contradiction. First, by Theorem~\ref{thm:local_min_gradient_zero}, we may assume $\nabla f(x) = 0$.

  Suppose $\langle \nabla^2 f(x)p,p\rangle<0$ for some $p \neq 0$. Define $H(u):=\langle \nabla^2 f(x+u p)p,p\rangle$.
  Then $H(0) = \langle \nabla^2 f(x)p,p\rangle < 0$. By continuity of the Hessian (Lemma~\ref{lem:continuous_hessian_line}),
  there exists $\delta>0$ such that $H(u) < H(0)/2 < 0$ for all $u \in (-\delta, \delta)$.

  But $x$ is a local minimizer, so $f(x+\alpha p) \ge f(x)$ for small $\alpha > 0$. By the exact second-order Taylor
  formula (Lemma~\ref{lem:taylor_second_order_exact}),
  \[
    f(x+\alpha p) - f(x) = \int_0^1\int_0^t \langle \nabla^2 f(x+s\alpha p)p,p\rangle\,ds\,dt
    = \alpha^2 \int_0^1\int_0^t \langle \nabla^2 f(x+s\alpha p)p,p\rangle\,ds\,dt.
  \]
  For small $\alpha > 0$, all points $x + s\alpha p$ lie in the ball where $H < 0$, so both integrals are negative,
  making $f(x+\alpha p) - f(x) < 0$, contradicting local minimality.
\end{proof}

\begin{definition}[Strict local minimizer]\label{def:IsStrictLocalMin}
  \lean{Lean4ML.Optimization.IsStrictLocalMin}
  \leanok{}
  \uses{}
  Lean definition:
  $\exists U\in \mathcal{N}(x),\ \forall y\in U,\ y\neq x \to f(x)<f(y)$.
\end{definition}

\begin{lemma}[Taylor lower bound from uniform Hessian lower bound]\label{lem:taylor_lower_bound_from_hessian_lb}
  \lean{Lean4ML.Optimization.taylor_lower_bound_from_hessian_lb}
  \leanok{}
  \uses{lem:taylor_second_order_exact,lem:continuous_hessian_line,lem:intervalIntegral_mono_of_continuous}
  Under a uniform bound $\varepsilon\|p\|^2 \le \langle \nabla^2 f(y)p,p\rangle$ on a ball around $x$
  and assuming $\nabla f(x)=0$, Lean proves
  $(\varepsilon/2)\|p\|^2 \le f(x+p)-f(x)$ for $\|p\|<\rho$ by integrating the pointwise bound inside the
  double-integral remainder formula.
\end{lemma}

\begin{proof}
  Start from the exact Taylor formula (Lemma~\ref{lem:taylor_second_order_exact}):
  \[
    f(x+p)-f(x)=\int_0^1\int_0^t \langle \nabla^2 f(x+s p)p,p\rangle\,ds\,dt.
  \]

  By convexity (or simply by norm estimates), the point $x+ sp$ stays within the ball $B(x,\rho)$ for all $s \in [0,1]$ when $\|p\| < \rho$.
  Within this ball, the Hessian quadratic form is bounded below: $\langle \nabla^2 f(y)p,p\rangle \ge \varepsilon\|p\|^2$.

  Therefore,
  \[
    f(x+p)-f(x) = \int_0^1\int_0^t \langle \nabla^2 f(x+s p)p,p\rangle\,ds\,dt
    \ge \int_0^1\int_0^t \varepsilon\|p\|^2\,ds\,dt
    = \varepsilon\|p\|^2 \int_0^1 t\,dt = \frac{\varepsilon}{2}\|p\|^2,
  \]
  as claimed.
\end{proof}

\begin{lemma}[Coercivity from positive definiteness (finite-dimensional)]\label{lem:inner_coercive_of_pd}
  \lean{Lean4ML.Optimization.inner_coercive_of_pd}
  \leanok{}
  \uses{}
  In finite dimensions, a positive definite continuous linear map $H$ satisfies
  $\exists \varepsilon>0,\ \forall p,\ \varepsilon\|p\|^2 \le \langle Hp,p\rangle$.
  Lean uses compactness of the sphere and \texttt{IsCompact.exists\_isLeast}.
\end{lemma}

\begin{lemma}[Continuity of Hessian map]\label{lem:continuous_hessian}
  \lean{Lean4ML.Optimization.continuous_hessian}
  \leanok{}
  \uses{def:hessian}
  If $f\in C^2$, then $y\mapsto \nabla^2 f(y)$ is continuous (Lean: \texttt{continuous\_fderiv} applied to $\nabla f$).
\end{lemma}

\begin{lemma}[Positive definite Hessian $\Rightarrow$ uniform lower bound on a ball]\label{lem:hessian_pd_local_lower_bound}
  \lean{Lean4ML.Optimization.hessian_pd_local_lower_bound}
  \leanok{}
  \uses{lem:inner_coercive_of_pd,lem:continuous_hessian,lem:inner_le_opNorm_mul_norm_sq}
  In finite dimensions, if $\nabla^2 f(x)$ is positive definite, then there exist $\varepsilon>0$ and $\rho>0$ such that
  $\varepsilon\|p\|^2 \le \langle \nabla^2 f(y)p,p\rangle$ for all $y\in B(x,\rho)$ and all $p$.
\end{lemma}

\begin{proof}
  Since $\nabla^2 f(x)$ is positive definite, Lemma~\ref{lem:inner_coercive_of_pd} provides $\varepsilon_0 > 0$ such that
  \[
    \varepsilon_0 \|p\|^2 \le \langle \nabla^2 f(x) p, p \rangle \quad \text{for all } p.
  \]

  By Lemma~\ref{lem:continuous_hessian}, the map $y \mapsto \nabla^2 f(y)$ is continuous. Therefore, we can find $\rho > 0$ such that
  \[
    \|\nabla^2 f(y) - \nabla^2 f(x)\| < \frac{\varepsilon_0}{2} \quad \text{for all } y \in B(x,\rho).
  \]

  For $y \in B(x,\rho)$ and any $p$, the operator norm inequality gives
  \[
    \left| \langle (\nabla^2 f(y) - \nabla^2 f(x))p, p \rangle \right| \le \|\nabla^2 f(y) - \nabla^2 f(x)\| \|p\|^2 < \frac{\varepsilon_0}{2} \|p\|^2.
  \]

  Therefore,
  \[
    \langle \nabla^2 f(y)p, p\rangle = \langle \nabla^2 f(x) p, p \rangle + \langle (\nabla^2 f(y) - \nabla^2 f(x))p, p \rangle
    > \varepsilon_0 \|p\|^2 - \frac{\varepsilon_0}{2} \|p\|^2 = \frac{\varepsilon_0}{2} \|p\|^2,
  \]
  which completes the proof.
\end{proof}

\begin{theorem}[Strict local minimizer from uniform Hessian lower bound]\label{thm:strict_local_min_of_gradient_zero_hessian_uniform_lb}
  \lean{Lean4ML.Optimization.strict_local_min_of_gradient_zero_hessian_uniform_lb}
  \leanok{}
  \uses{def:IsStrictLocalMin,lem:taylor_lower_bound_from_hessian_lb}
  If $f\in C^2$, $\nabla f(x)=0$, and the Hessian has a uniform positive lower bound on a ball around $x$, then $x$ is a strict local minimizer.
\end{theorem}

\begin{proof}
  Let $U = B(x, \rho)$. For any $y \in U$ with $y \neq x$, set $p = y - x$. Then $\|p\| < \rho$ and
  $f(y) - f(x) = f(x+p) - f(x)$.

  By Lemma~\ref{lem:taylor_lower_bound_from_hessian_lb} (applied to the uniform lower bound on $\nabla^2 f$
  over the ball around $x$), we have
  \[
    f(x+p) - f(x) \ge \frac{\varepsilon}{2} \|p\|^2.
  \]

  Since $p \neq 0$, we have $\|p\|^2 > 0$, so $f(y) > f(x)$. This holds for all $y \in U$ with $y \neq x$,
  showing that $x$ is a strict local minimizer.
\end{proof}

\begin{theorem}[Second-order sufficient condition (finite-dimensional)]\label{thm:second_order_sufficient_condition}
  \lean{Lean4ML.Optimization.second_order_sufficient_condition}
  \leanok{}
  \uses{lem:hessian_pd_local_lower_bound,thm:strict_local_min_of_gradient_zero_hessian_uniform_lb}
  If $f\in C^2$, $\nabla f(x)=0$, and $\nabla^2 f(x)$ is positive definite, then $x$ is a strict local minimizer.
\end{theorem}

\begin{proof}
  By hypothesis, $f \in C^2$ and $\nabla f(x) = 0$. The Hessian $\nabla^2 f(x)$ is positive definite.

  Lemma~\ref{lem:hessian_pd_local_lower_bound} guarantees that there exist $\varepsilon > 0$ and $\rho > 0$ such that
  \[
    \varepsilon \|p\|^2 \le \langle \nabla^2 f(y) p, p \rangle \quad \text{for all } y \in B(x, \rho) \text{ and all } p.
  \]

  With this uniform lower bound in place, Theorem~\ref{thm:strict_local_min_of_gradient_zero_hessian_uniform_lb}
  applies immediately to conclude that $x$ is a strict local minimizer.
\end{proof}

\chapter{GradientFlow}\label{sec:GradientFlow}

Throughout this section, $E$ is a real Hilbert space, $f : E \to \mathbb{R}$,
and $\nabla f$ denotes the (Fr\'echet) gradient in the Hilbert space structure.

% --------------------------------------------------------------------
% Definition: bounded-time flow
% --------------------------------------------------------------------

\begin{definition}[Gradient flow on a bounded interval]\label{def:GradientFlow}
  \lean{Lean4ML.Optimization.GradientFlow}
  \leanok{}
  \uses{}
  A \emph{bounded-time} gradient flow is a certificate that the ODE
  \[
    \dot x(t) = -\nabla f(x(t)),\qquad x(0)=x_0
  \]
  has a solution on $[0,T]$.
  In Lean, a \texttt{GradientFlow f x₀ T} is a structure with fields:
  \begin{itemize}
    \item \texttt{trajectory : $\mathbb{R}\to E$},
    \item \texttt{init : trajectory 0 = x₀},
    \item \texttt{hasDerivAt : ∀ t ∈ Icc 0 T, HasDerivAt trajectory (-(gradient f (trajectory t))) t}.
  \end{itemize}
\end{definition}

\begin{lemma}[Initial value]\label{lem:GradientFlow.trajectory_zero}
  \lean{Lean4ML.Optimization.GradientFlow.trajectory_zero}
  \leanok{}
  \uses{def:GradientFlow}
  For $\varphi : \texttt{GradientFlow f x₀ T}$, we have $\varphi(0)=x_0$.
\end{lemma}

\begin{lemma}[Differentiability of the trajectory on $\lbrack 0,T\rbrack$]\label{lem:GradientFlow.differentiableAt}
  \lean{Lean4ML.Optimization.GradientFlow.differentiableAt}
  \leanok{}
  \uses{def:GradientFlow}
  For $t\in[0,T]$, the map $t\mapsto \varphi(t)$ is differentiable at $t$.
  (Lean: \texttt{DifferentiableAt ℝ φ.trajectory t}.)
\end{lemma}

\begin{lemma}[Derivative equals negative gradient]\label{lem:GradientFlow.deriv_eq}
  \lean{Lean4ML.Optimization.GradientFlow.deriv_eq}
  \leanok{}
  \uses{def:GradientFlow,lem:GradientFlow.differentiableAt}
  For $t\in[0,T]$,
  \[
    \varphi'(t) = -\nabla f(\varphi(t)).
  \]
  (Lean: \texttt{deriv φ.trajectory t = -(gradient f (φ.trajectory t))}.)
\end{lemma}

% --------------------------------------------------------------------
% Energy dissipation along the flow
% --------------------------------------------------------------------

\begin{lemma}[Energy derivative along the flow]\label{lem:GradientFlow.hasDerivAt_energy}
  \lean{Lean4ML.Optimization.GradientFlow.hasDerivAt_energy}
  \leanok{}
  \uses{def:GradientFlow}
  If $f\in C^1$ and $\varphi$ is a gradient flow, then for $t\in[0,T]$
  \[
    \frac{d}{dt}\, f(\varphi(t)) = -\|\nabla f(\varphi(t))\|^2.
  \]

  By the chain rule, the derivative of the composition $f \circ \varphi$ at time $t$ is
  \[
    \frac{d}{dt} f(\varphi(t)) = Df(\varphi(t)) \cdot \varphi'(t) = \langle \nabla f(\varphi(t)), \varphi'(t) \rangle.
  \]

  Substituting the ODE $\varphi'(t) = -\nabla f(\varphi(t))$ gives
  \[
    \frac{d}{dt} f(\varphi(t)) = \langle \nabla f(\varphi(t)), -\nabla f(\varphi(t)) \rangle = -\|\nabla f(\varphi(t))\|^2,
  \]
  as claimed.
\end{lemma}

\begin{lemma}[Energy is non-increasing]\label{lem:GradientFlow.energy_antitone}
  \lean{Lean4ML.Optimization.GradientFlow.energy_antitone}
  \leanok{}
  \uses{lem:GradientFlow.hasDerivAt_energy}
  If $0\le r\le s\le T$, then
  \[
    f(\varphi(s)) \le f(\varphi(r)).
  \]

  Suppose for contradiction that $f(\varphi(s)) > f(\varphi(r))$. By the mean value theorem, there exists $t \in (r, s)$
  such that
  \[
    \varphi'(t) = \frac{f(\varphi(s)) - f(\varphi(r))}{s - r} > 0.
  \]
  But by Lemma~\ref{lem:GradientFlow.hasDerivAt_energy}, we have
  \[
    \varphi'(t) = -\|\nabla f(\varphi(t))\|^2 \le 0,
  \]
  a contradiction. Thus $f(\varphi(s)) \le f(\varphi(r))$.
\end{lemma}

\begin{definition}[Piecewise trajectory]\label{def:piecewiseTrajectory}
  \lean{Lean4ML.Optimization.piecewiseTrajectory}

\begin{definition}[Extend (chain) two gradient flows]

\chapter{Convexity}\label{sec:convexity}

Throughout this section, $E$ is a real normed space, $f : E \to \mathbb{R}$,
$s \subseteq E$ is a convex set, and $x, y \in E$.

% --------------------------------------------------------------------
% Definition of convexity
% --------------------------------------------------------------------

\begin{definition}[Convex function on a set]\label{def:convex_fn}
  \lean{ConvexOn}
  \leanok{}
  \uses{}
  We say that $f$ is \emph{convex on $s$} if $s$ is a convex set and for all $x, y \in s$
  and $a, b \ge 0$ with $a + b = 1$:
  \[
    f(a x + b y) \;\le\; a\,f(x) + b\,f(y).
  \]
\end{definition}

% --------------------------------------------------------------------
% Theorem 2.6
% --------------------------------------------------------------------

\begin{theorem}[Local minimizer on convex set is global]\label{thm:local_min_is_global_convex}
  \lean{IsMinOn.of_isLocalMinOn_of_convexOn}
  \leanok{}
  \uses{}
  If $f$ is convex on $s$ and $x \in s$ is a local minimizer of $f$ restricted to $s$,
  then $x$ is a global minimizer of $f$ on $s$.
\end{theorem}

\begin{theorem}[Set of minimizers is convex]\label{thm:minimizers_set_convex}
  \lean{Lean4ML.Optimization.ConvexOn.convex_minimizers_mathlib}
  \leanok{}
  \uses{}
  If $f$ is convex on $s$, then the set of minimizers
  \[
    \mathcal{M} := \{x \in s \mid \forall y \in s,\ f(x) \le f(y)\}
  \]
  is convex.
\end{theorem}

\begin{proof}
  Let $x, y \in \mathcal{M}$ and $a, b \ge 0$ with $a + b = 1$. Set $z := ax + by$.
  Since $s$ is convex, $z \in s$. For any $w \in s$, convexity of $f$ gives
  \[
    f(z) \le a\,f(x) + b\,f(y).
  \]
  Since $x$ and $y$ are both minimizers, $f(x) \le f(w)$ and $f(y) \le f(w)$, so
  \[
    f(z) \le a\,f(x) + b\,f(y) \le a\,f(w) + b\,f(w) = (a+b)\,f(w) = f(w).
  \]
  Hence $z \in \mathcal{M}$. If $\mathcal{M}$ is empty, it is trivially convex.
\end{proof}

% --------------------------------------------------------------------
% First-order characterization
% --------------------------------------------------------------------

\begin{lemma}[First-order characterization of convexity]\label{lem:convex_first_order_char}
  \lean{Lean4ML.Optimization.ConvexOn.add_fderiv_le}
  \leanok{}
  \uses{}
  If $f$ is convex on $s$ and Fr\'echet differentiable at $x \in s$, then for all $y \in s$:
  \[
    f(x) + Df(x)(y - x) \le f(y).
  \]
\end{lemma}

\begin{proof}
  \textbf{Step 1: Convexity upper bound along the segment.}
  For $\alpha \in [0,1]$, the convexity of $f$ gives
  \[
    f(x + \alpha(y-x)) = f((1-\alpha)x + \alpha y) \le (1-\alpha)f(x) + \alpha f(y).
  \]

  \textbf{Step 2: Rearranging into a difference quotient.}
  For $\alpha \in (0,1]$, subtracting $f(x)$ and dividing by $\alpha > 0$ yields
  \[
    \frac{f(x + \alpha(y-x)) - f(x)}{\alpha} \le f(y) - f(x).
  \]

  \textbf{Step 3: Taking the limit.}
  By the chain rule, the map $\alpha \mapsto f(x + \alpha(y-x))$ has derivative $Df(x)(y-x)$ at $\alpha = 0$.
  Therefore the difference quotients on the left converge to $Df(x)(y-x)$ as $\alpha \to 0^+$.
  Passing the limit through the inequality gives
  \[
    Df(x)(y-x) \le f(y) - f(x),
  \]
  which rearranges to $f(x) + Df(x)(y-x) \le f(y)$.
\end{proof}

\begin{theorem}[Zero derivative implies global minimizer]\label{thm:zero_deriv_implies_global_min}
  \lean{Lean4ML.Optimization.ConvexOn.isMinOn_of_hasDerivAt_eq_zero}
  \leanok{}
  \uses{lem:convex_first_order_char}
  If $f$ is convex on $s$, $x \in s$, and $Df(x) = 0$, then $x$ is a global minimizer of $f$ on $s$.
\end{theorem}

\begin{proof}
  Apply Lemma~\ref{lem:convex_first_order_char} with the given $x$ and any $y \in s$:
  \[
    f(x) + Df(x)(y-x) \le f(y).
  \]
  Since $Df(x) = 0$, the second term vanishes, giving $f(x) \le f(y)$ for all $y \in s$.
\end{proof}

\chapter{Polyak--Łojasiewicz condition and convergence}\label{sec:PL}

\begin{lemma}[Energy gap derivative and PL differential inequality]\label{lem:energyGap_hasDerivAt_and_le}
  \lean{Lean4ML.Optimization.energyGap_hasDerivAt_and_le}
  \leanok{}
  \uses{def:GradientFlowGlobal,def:GloballyPL,lem:GradientFlow.hasDerivAt_energy}
  Let $g(t):=f(\varphi(t))-\inf f$, where $\varphi$ is a global gradient flow.
  If $f\in C^1$ and $f$ satisfies the global $\mu$-PL condition, then for every $t\ge 0$,
  \[
    g'(t)=-\|\nabla f(\varphi(t))\|^2
    \quad\text{and}\quad
    g'(t)\le -2\mu\,g(t).
  \]
\end{lemma}

\begin{proof}
  Differentiate $g(t)=f(\varphi(t))-\inf f$. The constant $\inf f$ contributes no derivative, and the chain rule
  along the flow gives
  \[
    g'(t)=\frac{d}{dt}f(\varphi(t))=-\|\nabla f(\varphi(t))\|^2.
  \]
  The global PL inequality gives
  \[
    \|\nabla f(\varphi(t))\|^2\ge 2\mu\bigl(f(\varphi(t))-\inf f\bigr)=2\mu\,g(t),
  \]
  so multiplying by $-1$ yields $g'(t)\le -2\mu\,g(t)$.
\end{proof}

\begin{lemma}[Exponential decay of the energy gap]\label{lem:energyGap_exp_bound}
  \lean{Lean4ML.Optimization.energyGap_exp_bound}
  \leanok{}
  \uses{lem:energyGap_hasDerivAt_and_le}
  Under the same hypotheses, the energy gap satisfies
  \[
    f(\varphi(t)) - \inf f \le (f(x_0)-\inf f)\,e^{-2\mu t}
  \]
  for all $t\ge 0$.
\end{lemma}

\begin{proof}
  Consider the auxiliary function
  \[
    h(t):=\bigl(f(\varphi(t))-\inf f\bigr)e^{2\mu t}.
  \]
  Using the previous lemma and the product rule, one finds that $h'(t)\le 0$ for $t\ge 0$, so $h$ is non-increasing.
  Therefore $h(t)\le h(0)=f(x_0)-\inf f$, and multiplying by $e^{-2\mu t}$ gives the stated estimate.
\end{proof}

\begin{theorem}[Convergence of the gradient flow under the PL condition]\label{thm:gradientFlow_PL_tendsto}
  \lean{Lean4ML.Optimization.gradientFlow_PL_tendsto}
  \leanok{}
  \uses{def:GloballyPL,lem:energyGap_exp_bound}
  Assume in addition that $f$ is bounded below. Then along any global gradient flow,
  \[
    f(\varphi(t)) \to \inf f \quad \text{as } t\to\infty.
  \]
\end{theorem}

\begin{proof}
  The global PL estimate gives the upper bound from Lemma~\ref{lem:energyGap_exp_bound}. Since $f$ is bounded below,
  we also have $f(\varphi(t)) - \inf f\ge 0$ for all $t\ge 0$.
  The right-hand side is an exponentially decaying term, hence tends to $0$ as $t\to\infty$.
  By squeezing, the energy gap converges to $0$, which is equivalent to $f(\varphi(t))\to \inf f$.
\end{proof}


\chapter{Strong Convexity}\label{sec:strong_convexity}

Throughout this section, $E$ is a real Hilbert space, $f : E \to \mathbb{R}$,
$s \subseteq E$ is a convex set, $m > 0$, and $x, y \in E$.

% --------------------------------------------------------------------
% Definition of strong convexity
% --------------------------------------------------------------------

\begin{definition}[Strongly convex function on a set]\label{def:strong_convex_fn}
  \lean{StrongConvexOn}
  \leanok{}
  \uses{def:convex_fn}
  We say that $f$ is \emph{strongly convex on $s$ with modulus $m > 0$} if $s$ is a convex set
  and for all $x, y \in s$ and $a, b \ge 0$ with $a + b = 1$:
  \[
    f(a x + b y) \;\le\; a\,f(x) + b\,f(y) - \tfrac{m}{2}\,ab\,\|x - y\|^2.
  \]
  In particular, taking $a = 1-\alpha$, $b = \alpha$ recovers the modulated convexity inequality
  \[
    f\bigl((1-\alpha)x + \alpha y\bigr) \;\le\; (1-\alpha)f(x) + \alpha f(y) - \tfrac{m}{2}\alpha(1-\alpha)\|x-y\|^2.
  \]
\end{definition}

% --------------------------------------------------------------------
% Reduction to convexity
% --------------------------------------------------------------------

\begin{lemma}[Strong convexity reduces to convexity]\label{lem:strongConvexOn_iff_convex}
  \lean{strongConvexOn_iff_convex}
  \leanok{}
  \uses{}
  $f$ is strongly convex on $s$ with modulus $m$ if and only if
  $x \mapsto f(x) - \dfrac{m}{2}\|x\|^2$ is convex on $s$:
  \[
    \mathrm{StrongConvexOn}\ s\ m\ f
    \;\iff\;
    \mathrm{ConvexOn}\ s\ \Bigl(x \mapsto f(x) - \tfrac{m}{2}\|x\|^2\Bigr).
  \]
\end{lemma}

% --------------------------------------------------------------------
% First-order lower bound for strongly convex functions
% --------------------------------------------------------------------

\begin{lemma}[First-order lower bound for strongly convex functions]\label{lem:strong_convex_lower_bound}
  \lean{Lean4ML.Optimization.StronglyConvexOn.add_fderiv_le}
  \leanok{}
  \uses{lem:strongConvexOn_iff_convex,lem:convex_first_order_char}
  If $f$ is strongly convex on $s$ with modulus $m > 0$ and Fr\'echet differentiable at $x \in s$,
  then for all $y \in s$:
  \[
    f(y) \ge f(x) + Df(x)(y - x) + \frac{m}{2}\|y - x\|^2.
  \]
\end{lemma}

\begin{proof}
  Define $g : E \to \mathbb{R}$ by $g(z) = f(z) - \frac{m}{2}\|z\|^2$.
  By Lemma~\ref{lem:strongConvexOn_iff_convex}, $g$ is convex on $s$.
  Since $f$ is Fr\'echet differentiable at $x$ and $z \mapsto \frac{m}{2}\|z\|^2$ is
  differentiable everywhere, $g$ is Fr\'echet differentiable at $x$ with
  \[
    Dg(x) = Df(x) - D\!\left(\tfrac{m}{2}\|\cdot\|^2\right)(x).
  \]
  A direct computation gives $D\!\left(\frac{m}{2}\|\cdot\|^2\right)(x)(v) = m\langle x, v\rangle$
  for all $v$.
  Apply Lemma~\ref{lem:convex_first_order_char} to the convex function $g$:
  \[
    g(x) + Dg(x)(y - x) \le g(y).
  \]
  Substituting $g(z) = f(z) - \frac{m}{2}\|z\|^2$ and $Dg(x)(v) = Df(x)(v) - m\langle x, v\rangle$:
  \[
    f(x) - \tfrac{m}{2}\|x\|^2
    + Df(x)(y-x) - m\langle x,\, y-x\rangle
    \;\le\;
    f(y) - \tfrac{m}{2}\|y\|^2.
  \]
  Rearranging, it remains to show that
  \[
    -\tfrac{m}{2}\|x\|^2 - m\langle x,\, y-x\rangle + \tfrac{m}{2}\|y\|^2
    \;=\; \tfrac{m}{2}\|y - x\|^2.
  \]
  Indeed, expanding $\langle x, y-x\rangle = \langle x,y\rangle - \|x\|^2$:
  \begin{align*}
    -\tfrac{m}{2}\|x\|^2 - m\langle x,y\rangle + m\|x\|^2 + \tfrac{m}{2}\|y\|^2
    &= \tfrac{m}{2}\|x\|^2 - m\langle x,y\rangle + \tfrac{m}{2}\|y\|^2 \\
    &= \tfrac{m}{2}\bigl(\|x\|^2 - 2\langle x,y\rangle + \|y\|^2\bigr)
     = \tfrac{m}{2}\|y - x\|^2.
  \end{align*}
  Therefore $f(x) + Df(x)(y-x) + \frac{m}{2}\|y-x\|^2 \le f(y)$.
\end{proof}

\begin{theorem}[Unique global minimizer under strong convexity]\label{thm:unique_minimizer_strong_convex}
  \lean{Lean4ML.Optimization.StrongConvexOn.isMinOn_of_fderiv_eq_zero}
  \leanok{}
  \uses{lem:strong_convex_lower_bound,def:strong_convex_fn}
  Suppose that $f$ is strongly convex on $s$ with modulus $m > 0$ and Fr\'echet
  differentiable at $x^* \in s$. If $Df(x^*) = 0$, then $x^*$ is a global
  minimizer of $f$ on $s$, and any other global minimizer $y \in s$ satisfies
  $y = x^*$.
\end{theorem}

\begin{proof}
Apply Lemma~\ref{lem:strong_convex_lower_bound} with $x = x^*$ and any $y$ in the domain:
$$f(y) \geq f(x^*) + Df(x^*)(y-x^*) + \frac{m}{2}\|y-x^*\|^2.$$
Since $x^*$ is a stationary point, $Df(x^*) = 0$, so:
$$f(y) \geq f(x^*) + \frac{m}{2}\|y-x^*\|^2.$$
By the definition of strong convexity, the modulus $m > 0$. Furthermore, for any $y \neq x^*$, the squared norm $\|y-x^*\|^2 > 0$.
Therefore, for all $y \neq x^*$, the term $\frac{m}{2}\|y-x^*\|^2$ is strictly positive, which implies:
$$f(y) > f(x^*)$$
This demonstrates that $x^*$ is the strictly unique global minimizer of $f$.
\end{proof}

\begin{theorem}[Stationary point is global minimizer and matches the infimum]\label{thm:isMinOn_of_stationary}
  \lean{Lean4ML.Optimization.StrongConvexOn.isMinOn_of_stationary}
  \leanok{}
  \uses{thm:unique_minimizer_strong_convex}
  If $f\in C^1$ is strongly convex on $E$ with modulus $m>0$ and $x$ is a stationary point
  (i.e. $Df(x)=0$), then $x$ is the global minimizer and $\inf f = f(x)$.
\end{theorem}
\begin{proof}
  \leanok{}
  Apply Theorem~\ref{thm:unique_minimizer_strong_convex} to see that any stationary point is a global minimizer.
  Since a minimizer's value is a lower bound of $f$, and the infimum is the greatest lower bound, equality follows
  by the usual bounding arguments (show $f(x)\le iInf f$ and $iInf f \le f(x)$ using bounded-below).
\end{proof}

\begin{lemma}[Midpoint distance bound]\label{lem:midpoint_dist_bound}
  \lean{Lean4ML.Optimization.StrongConvexOn.midpoint_dist_bound}
  \leanok{}
  \uses{}
  If $f$ is strongly convex on $E$ with modulus $\mu>0$ and $f$ is bounded below, then for any $x,y\in E$,
  \[
    \frac{\mu}{8}\,\|x-y\|^2 \le \frac{f(x)+f(y)}{2} - \inf f.
  \]
\end{lemma}
\begin{proof}
  \leanok{}
  Apply the strong convexity inequality at the midpoint $\tfrac12(x+y)$ (take $a=b=1/2$) to get an inequality
  relating $f(\frac{x+y}{2})$, $f(x),f(y)$ and $\mu\|x-y\|^2$. Using $\inf f\le f(\frac{x+y}{2})$ and simple
  algebra produces the claimed bound.
\end{proof}

\begin{theorem}[Existence of a minimizer for C¹ strongly convex functions]\label{thm:exists_minimizer_strong_convex}
  \lean{Lean4ML.Optimization.StrongConvexOn.exists_minimizer}
  \leanok{}
  \uses{lem:midpoint_dist_bound}
  If $f\in C^1$ is strongly convex on a complete Hilbert space $E$ with modulus $\mu>0$, then $f$ attains its infimum;
  i.e. there exists $x^*\in E$ with $f(x^*)=\inf f$.
\end{theorem}
\begin{proof}
  \leanok{}
  Use a minimizing sequence $(x_n)$ with $f(x_n)\downarrow\inf f$. The midpoint distance bound
  (Lemma~\ref{lem:midpoint_dist_bound}) gives a bound of the form $\|x_n-x_m\|^2\le C( f(x_n)+f(x_m) - 2\inf f)$,
  which tends to $0$ as $n,m\to\infty$, so $(x_n)$ is Cauchy. Completeness yields a limit $x^*$, and continuity of $f$
  (from $C^1$) implies $f(x^*)=\inf f$.
\end{proof}

\begin{lemma}[Quadratic growth at a minimizer]\label{lem:quadratic_growth_at_minimizer}
  \lean{Lean4ML.Optimization.StrongConvexOn.quadratic_growth_at_minimizer}
  \leanok{}
  \uses{thm:isMinOn_of_stationary}
  If $f\in C^1$ is strongly convex on $E$ with modulus $\mu>0$ and $x^*$ is a minimizer, then for all $y\in E$:
  \[
    \frac{\mu}{2}\,\|y-x^*\|^2 \le f(y)-f(x^*).
  \]
\end{lemma}
\begin{proof}
  \leanok{}
  At a minimizer $x^*$ the derivative vanishes, so applying the first-order strong convexity bound
  (Lemma~\ref{lem:strong_convex_lower_bound}) with $x=x^*$ yields the inequality after cancelling the derivative term.
\end{proof}

\begin{lemma}[$C^1$ regularity of the gradient]\label{lem:gradient_contDiff_one}
  \lean{Lean4ML.Optimization.gradient_contDiff_one}
  \leanok{}
  \uses{}
  If $f \in C^2$, then $y \mapsto \nabla f(y)$ is in $C^1$.

\begin{lemma}[Hessian as Fr\'echet derivative of the gradient]\label{lem:hasFDerivAt_gradient}
  \lean{Lean4ML.Optimization.hasFDerivAt_gradient}
  \leanok{}
  \uses{def:hessian,lem:gradient_contDiff_one}
  If $f \in C^2$, then at every $x \in E$, the map $y \mapsto \nabla f(y)$ has Fr\'echet
  derivative $\nabla^2 f(x)$.
\end{lemma}

\begin{lemma}[Strong convexity $\Rightarrow$ Hessian lower bound]\label{lem:strongConvexOn_imp_hessian_lb}
  \lean{Lean4ML.Optimization.strongConvexOn_imp_hessian_lb}
  \leanok{}
  \uses{def:strong_convex_fn,def:hessian,lem:hasFDerivAt_gradient}
  If $f \in C^2$ is strongly convex on $E$ with modulus $m$, then
  \[
    m\|u\|^2 \;\le\; \langle \nabla^2 f(x)\, u,\, u\rangle \quad \text{for all } x, u \in E.
  \]
\end{lemma}

\begin{proof}
  Fix $x, u$ and $t > 0$. The first-order strong convexity bound
  (Lemma~\ref{lem:strong_convex_lower_bound}) at the pair $(x, x + tu)$ and at
  $(x+tu, x)$ together with summation yields
  \[
    \frac{(Df(x+tu) - Df(x))(u)}{t} \;\ge\; m\|u\|^2.
  \]
  As $t \to 0^+$, the left-hand side converges to $\langle \nabla^2 f(x)u, u\rangle$
  (using Lemma~\ref{lem:hasFDerivAt_gradient}), giving the claim.
\end{proof}

\begin{lemma}[Hessian lower bound $\Rightarrow$ strong convexity]\label{lem:hessian_lb_imp_strongConvexOn}
  \lean{Lean4ML.Optimization.hessian_lb_imp_strongConvexOn}
  \leanok{}
  \uses{def:strong_convex_fn,def:hessian,lem:strongConvexOn_iff_convex,lem:hasFDerivAt_gradient}
  If $f \in C^2$ and $m\|u\|^2 \le \langle \nabla^2 f(x) u, u\rangle$ for all
  $x, u \in E$, then $f$ is strongly convex on $E$ with modulus $m$.
\end{lemma}

\begin{proof}
  By Lemma~\ref{lem:strongConvexOn_iff_convex}, it suffices to show that
  $g(y) := f(y) - \frac{m}{2}\|y\|^2$ is convex on $E$. For any line restriction
  $\varphi(t) := g(x + t d)$, a direct computation gives
  $\varphi''(t) = \langle \nabla^2 f(x+td)\, d, d\rangle - m\|d\|^2 \ge 0$,
  so $\varphi$ is convex. Since this holds for every line, $g$ is convex.
\end{proof}

\begin{lemma}[Hessian characterization of strong convexity]\label{lem:hessian_char_strong_convex}
  \lean{Lean4ML.Optimization.strongConvexOn_iff_hessian_lb}
  \leanok{}
  \uses{lem:strongConvexOn_imp_hessian_lb,lem:hessian_lb_imp_strongConvexOn}
  Suppose $f \in C^2$. Then $f$ is strongly convex on $E$ with modulus $m$ if and only if
  $m\|u\|^2 \le \langle \nabla^2 f(x) u, u\rangle$ for all $x, u \in E$, i.e.\ $\nabla^2 f(x) \succeq mI$.
\end{lemma}

\begin{proof}
  Direct from Lemmas~\ref{lem:strongConvexOn_imp_hessian_lb} and~\ref{lem:hessian_lb_imp_strongConvexOn}.
\end{proof}

\begin{lemma}[$L$-smoothness $\Rightarrow$ Hessian upper bound]\label{lem:lSmooth_imp_hessian_ub}
  \lean{Lean4ML.Optimization.lSmooth_imp_hessian_ub}

\begin{lemma}[Operator-norm bound for symmetric PSD operators with quadratic-form bound]\label{lem:opNorm_le_of_inner_le_of_symm_of_nonneg}
  \lean{Lean4ML.Optimization.opNorm_le_of_inner_le_of_symm_of_nonneg}
  \leanok{}
  \uses{}
  Let $A : E \to_L E$ be self-adjoint with $0 \le \langle A u, u\rangle$ and
  $\langle A u, u\rangle \le c\|u\|^2$ for all $u$, where $0 \le c$. Then $\|A\| \le c$.
\end{lemma}

\begin{proof}
  By the polarization identity (using self-adjointness),
  \[
    4\,\langle A u, v\rangle = \langle A(u+v), u+v\rangle - \langle A(u-v), u-v\rangle.
  \]
  Combining with $0 \le \langle A w, w\rangle \le c\|w\|^2$ gives
  $|\langle A u, v\rangle| \le \tfrac{c}{2}\,(\|u\|^2 + \|v\|^2)$.
  Substituting $v = (\|u\|/\|A u\|)\,A u$ yields $\|A u\| \le c\|u\|$, hence $\|A\| \le c$.
\end{proof}

\begin{lemma}[Self-adjointness of the Hessian]\label{lem:hessian_isSymmetric}
  \lean{Lean4ML.Optimization.hessian_isSymmetric}
  \leanok{}
  \uses{def:hessian}
  If $f \in C^2$, then for every $x, u, v \in E$:
  \[
    \langle \nabla^2 f(x)\, u,\, v\rangle = \langle u,\, \nabla^2 f(x)\, v\rangle.
  \]
\end{lemma}

\begin{proof}
  By the symmetry of the second Fr\'echet derivative
  (\texttt{ContDiffAt.isSymmSndFDerivAt}), $(D^2 f(x))(u)(v) = (D^2 f(x))(v)(u)$.
  Combining with the Riesz duality used to define the gradient ($\nabla f$ is
  the composition of $Df$ with $(\mathrm{InnerProductSpace.toDual})^{-1}$) and
  the chain rule transfers the symmetry to the Hessian.
\end{proof}

\begin{lemma}[Hessian upper bound $+$ convexity $\Rightarrow$ $L$-smoothness]\label{lem:hessian_ub_imp_lSmooth}
  \lean{Lean4ML.Optimization.hessian_ub_imp_lSmooth}
  \leanok{}
  \uses{def:l_smooth_fn,def:hessian,thm:hessianBoundImpliesLSmoothOnUniv,lem:hasFDerivAt_gradient,lem:opNorm_le_of_inner_le_of_symm_of_nonneg,lem:hessian_isSymmetric,lem:strongConvexOn_imp_hessian_lb}
  Suppose $f \in C^2$ is convex on $E$ and $\langle \nabla^2 f(x)\, u, u\rangle \le L\|u\|^2$
  for all $x, u$. Then $f$ is globally $L$-smooth.
\end{lemma}

\begin{proof}
  By Theorem~\ref{thm:hessianBoundImpliesLSmoothOnUniv}, it suffices to bound
  $\|\nabla^2 f(x)\| \le L$ for all $x$. The Hessian is self-adjoint by
  Lemma~\ref{lem:hessian_isSymmetric}. Since $f$ is convex, it is strongly
  convex with modulus $0$, so Lemma~\ref{lem:strongConvexOn_imp_hessian_lb}
  gives $\langle \nabla^2 f(x)\, u, u\rangle \ge 0$. Conclude by
  Lemma~\ref{lem:opNorm_le_of_inner_le_of_symm_of_nonneg}.
\end{proof}

\begin{corollary}[Hessian sandwich characterization of $L$-smooth convex functions]\label{cor:hessian_sandwich}
  \lean{Lean4ML.Optimization.convexOn_lSmoothOn_iff_hessian_sandwich}
  \leanok{}
  \uses{lem:lSmooth_imp_hessian_ub,lem:hessian_ub_imp_lSmooth}
  Suppose $f \in C^2$ is convex on $E$. Then $f$ is globally $L$-smooth if and only if
  $\langle \nabla^2 f(x)\, u, u\rangle \le L\|u\|^2$ for all $x, u \in E$, i.e.\
  $\nabla^2 f(x) \preceq LI$.
\end{corollary}

\begin{proof}
  Direct from Lemmas~\ref{lem:lSmooth_imp_hessian_ub} and~\ref{lem:hessian_ub_imp_lSmooth}.
\end{proof}

\begin{lemma}[Inner-product expansion of $\|a + r\,b\|^2$]\label{lem:norm_add_smul_sq}
  \lean{Lean4ML.Optimization.norm_add_smul_sq}
  \leanok{}
  \uses{}
  For $a, b \in E$ and $r \in \mathbb{R}$:
  \[
    \|a + r\,b\|^2 = \|a\|^2 + 2r\,\langle a, b\rangle + r^2\,\|b\|^2.
  \]
\end{lemma}

\begin{theorem}[Strong convexity $\Rightarrow$ global PL]\label{thm:StrongConvexOn_globallyPL}
  \lean{Lean4ML.Optimization.StrongConvexOn.globallyPL}
  \leanok{}
  \uses{def:strong_convex_fn,def:GloballyPL,lem:strong_convex_lower_bound,lem:norm_add_smul_sq}
  If $f$ is strongly convex on $E$ with modulus $m > 0$ and Fr\'echet
  differentiable on $E$, then $f$ is globally $m$-PL:
  \[
    \|\nabla f(x)\|^2 \;\ge\; 2m\,\bigl(f(x) - \inf f\bigr) \quad \text{for all } x \in E.
  \]
\end{theorem}

\begin{proof}
  Fix $x$ and set $g := \nabla f(x)$. By the first-order lower bound
  (Lemma~\ref{lem:strong_convex_lower_bound}), for every $z \in E$,
  \[
    f(z) \;\ge\; f(x) + \langle g,\, z - x\rangle + \tfrac{m}{2}\,\|z - x\|^2.
  \]
  Expanding $0 \le \|g + m(z-x)\|^2$ via Lemma~\ref{lem:norm_add_smul_sq} gives
  \[
    0 \;\le\; \|g\|^2 + 2m\,\langle g,\, z-x\rangle + m^2\,\|z-x\|^2.
  \]
  Multiplying the first inequality by $2m > 0$ and adding to the second yields
  \[
    f(x) - \tfrac{1}{2m}\,\|g\|^2 \;\le\; f(z) \quad \text{for all } z \in E.
  \]
  Hence $f$ is bounded below and $\inf f \ge f(x) - \tfrac{1}{2m}\|g\|^2$,
  which rearranges to $\|g\|^2 \ge 2m\,(f(x) - \inf f)$.
\end{proof}

\chapter{Gradient flow for strongly convex functions}\label{sec:GF_StronglyConvex}

In this chapter we collect the consequences of strong convexity for gradient
flows. The key observations are that strong convexity implies the Polyak-L\'ojasiewicz
(PL) inequality (hence exponential decay of the energy gap along the flow), and that
the trajectory itself converges to the unique global minimizer.

\begin{theorem}[Energy convergence under strong convexity]\label{thm:gradientFlow_strongConvex_tendsto}
  \lean{Lean4ML.Optimization.gradientFlow_strongConvex_tendsto}
  \leanok{}
  \uses{def:GloballyPL,Lean4ML.Optimization.StrongConvexOn.globallyPL,Lean4ML.Optimization.StrongConvexOn.bddBelow_range,Lean4ML.Optimization.gradientFlow_PL_tendsto}
  Let $f\in C^1$ be strongly convex on $E$ with modulus $\mu>0$, and let $\varphi$ be
  a global gradient flow for $f$. Then
  \[
    f(\varphi(t)) \to \inf f \quad\text{as } t\to\infty.
  \]
\end{theorem}
\begin{proof}
  \leanok{}
  Strong convexity (with $\mu>0$) together with differentiability implies the
  global PL inequality by Theorem~\ref{thm:StrongConvexOn_globallyPL} (Lean: \texttt{StrongConvexOn.globallyPL}).
  Moreover a $C^1$ strongly convex function is bounded below (see
  \texttt{StrongConvexOn.bddBelow_range}). Applying the convergence result
  for flows under the PL condition (Theorem~\ref{thm:gradientFlow_PL_tendsto})
  yields the claim.
\end{proof}

\begin{theorem}[Trajectory convergence to the unique minimizer]\label{thm:gradientFlow_strongConvex_trajectory_tendsto}
  \lean{Lean4ML.Optimization.gradientFlow_strongConvex_trajectory_tendsto}
  \leanok{}
  \uses{Lean4ML.Optimization.StrongConvexOn.exists_minimizer,Lean4ML.Optimization.StrongConvexOn.quadratic_growth_at_minimizer,Lean4ML.Optimization.gradientFlow_strongConvex_tendsto}
  Let $f\in C^1$ be strongly convex on $E$ with modulus $\mu>0$, and let $\varphi$ be
  a global gradient flow for $f$. Then there exists a unique minimizer $x^*$ of $f$ and
  the trajectory converges:
  \[
    \varphi(t) \to x^* \quad\text{as } t\to\infty.
  \]
\end{theorem}
\begin{proof}
  \leanok{}
  By strong convexity and $C^1$ regularity, $f$ attains a unique global minimizer
  $x^*$ (Lean: \texttt{StrongConvexOn.exists_minimizer} together with the uniqueness
  provided by \texttt{StrongConvexOn.isMinOn_of_fderiv_eq_zero}). From
  Theorem~\ref{thm:gradientFlow_strongConvex_tendsto} we know the energy along the
  trajectory satisfies $f(\varphi(t))\to f(x^*)=\inf f$.

  Strong convexity also yields quadratic growth around the minimizer: for all $t\ge 0$
  we have
  \[
    \frac{\mu}{2}\|\varphi(t)-x^*\|^2 \le f(\varphi(t)) - f(x^*).
  \]
  Thus the vanishing of the energy gap $f(\varphi(t))-f(x^*)\to 0$ forces
  $\|\varphi(t)-x^*\|^2\to 0$, hence $\varphi(t)\to x^*$ as claimed.
\end{proof}
